Vue.prototype.$http = axios

用法：
methods: {
      login () {
        this.$http({
          method: 'POST',
          url: '/controller/login/user/admin',
          data: {
            user_name: this.user,
            u_passwd: this.password
          }
        }).then(function (res) {
          if (res.data.status === true) {
            window.location.href = '#/index'
          }
        }).catch(function (res) {
          console.log(res)
        })
      }
    }
